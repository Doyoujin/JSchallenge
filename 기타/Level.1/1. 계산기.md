```JavaScript
const calculator={

    add: function (a,b){

        console.log(a+b);

    },

    minus: function(a,b){

        console.log(a-b);

    },

    divide: function(a,b){

        console.log(a/b);

    },

    powerOf: function(a,b){

        console.log(a**b);

    },

}

​

calculator.add(3,5);

calculator.minus(3,5);

calculator.divide(3,5);

calculator.powerOf(3,5);
```
